<script>

    const audio = document.getElementById('main-audio');

    const statusKnob = document.getElementById('status-knob');

    const seekKnob = document.getElementById('seek-knob');

    const volSteps = document.querySelectorAll('.vol-step');

    const statusText = document.getElementById('status-text');

 

    let isDraggingStatus = false;

    let isDraggingSeek = false;

 

    volSteps.forEach(step => {

        step.addEventListener('click', () => {

            volSteps.forEach(s => s.classList.remove('active'));

            step.classList.add('active');

           

            const vol = parseFloat(step.dataset.vol);

            audio.volume = vol;

            statusText.innerText = `VOLUME SET TO: ${vol * 100}%`;

        });

    });

 

    function getAngle(event, element) {

        const rect = element.getBoundingClientRect();

        const centerX = rect.left + rect.width / 2;

        const centerY = rect.top + rect.height / 2;

        const x = event.clientX - centerX;

        const y = event.clientY - centerY;

        let angle = Math.atan2(y, x) * (180 / Math.PI);

        return angle + 90;

    }

 

    statusKnob.addEventListener('mousedown', () => isDraggingStatus = true);

    window.addEventListener('mouseup', () => isDraggingStatus = false);

 

    window.addEventListener('mousemove', (e) => {

        if (!isDraggingStatus) return;

        const angle = getAngle(e, statusKnob);

        statusKnob.style.transform = `rotate(${angle}deg)`;

       

        if (angle > 20) {

            const speed = Math.min(2, Math.max(0.5, angle / 100));

            audio.playbackRate = speed;

            if (audio.paused) audio.play();

            statusText.innerText = `PLAYING: ${speed.toFixed(2)}x SPEED`;

        } else {

            audio.pause();

            statusText.innerText = `PAUSED / STOPPED`;

        }

    });

 

    seekKnob.addEventListener('mousedown', () => isDraggingSeek = true);

    window.addEventListener('mouseup', () => isDraggingSeek = false);

 

    window.addEventListener('mousemove', (e) => {

        if (!isDraggingSeek) return;

        const angle = getAngle(e, seekKnob);

        seekKnob.style.transform = `rotate(${angle}deg)`;

       

        const percent = (angle % 360) / 360;

        if (percent >= 0) {

            audio.currentTime = percent * audio.duration;

            statusText.innerText = `SEEKING: ${Math.floor(percent * 100)}%`;

        }

    });

 

    audio.volume = 0.5;

</script>
